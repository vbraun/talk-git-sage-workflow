\begin{frame}
  \frametitle{More Cool Stuff}
  
  \begin{block}{Sage+Git developer manual}
    The current draft for the Sage+Git development manual is here:
    \url{http://sagemath.github.io/git-developer-guide/}
  \end{block}

  \begin{itemize}
  \item \cmd{git help <command>} shows the help for any git command.
  \item \cmd{git reset} modifies the branch to point to an arbitrary
    commit. For example, used to abandon commits.
  \item \cmd{git stash} is a place to put changes temporarily
  \item \cmd{git reflog} history of your local git commands. Allows
    you to undo anything.
  \item Detached heads: \cmd{git checkout <sha1>} instead of \cmd{git
      checkout <branch>}
  \end{itemize}
\end{frame}
